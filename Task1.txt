Function GetPlayerName
	OUTPUT ""
	Continue <- True
	WHILE Continue == True
		OUTPUT "Please enter your name"
		PlayerName <- USERINPUT
		IF PlayerName != ""
			Continue <- False
		ENDIF
	ENDWHILE
	OUTPUT ""
	RETURN PlayerName
ENDFUNCION
